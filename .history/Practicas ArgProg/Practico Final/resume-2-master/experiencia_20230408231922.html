function updateCountdown() { // Calcs const NOW = new Date() const DURATION = DATE_TARGET - NOW; const REMAINING_DAYS = Math.floor(DURATION / MILLISECONDS_OF_A_DAY); const REMAINING_HOURS = Math.floor((DURATION % MILLISECONDS_OF_A_DAY) / MILLISECONDS_OF_A_HOUR); const REMAINING_MINUTES = Math.floor((DURATION % MILLISECONDS_OF_A_HOUR) / MILLISECONDS_OF_A_MINUTE); const REMAINING_SECONDS = Math.floor((DURATION % MILLISECONDS_OF_A_MINUTE) / MILLISECONDS_OF_A_SECOND); // Render SPAN_DAYS.textContent = REMAINING_DAYS; SPAN_HOURS.textContent = REMAINING_HOURS; SPAN_MINUTES.textContent = REMAINING_MINUTES; SPAN_SECONDS.textContent = REMAINING_SECONDS; } //=== // INIT //=== updateCountdown(); // Refresh every second setInterval(updateCountdown, MILLISECONDS_OF_A_SECOND); });
